void Bubl1(int Mass[], int n)
{
	int i, j;
	for (int i = 0; i < n; i++) {
		if (i % 2 == 0) {
			for (int j = 0; j < n - 1; j += 2)
			{
				if (Mass[j] > Mass[i + 1])swap(Mass[j], Mass[i + 1]);
			}
		}
		else {
			for (int j = 1; j < n - 1; j += 2)
			{
				if (Mass[j] > Mass[j + 1])swap(Mass[j], Mass[j + 1]);
			}

		}
	}
}

vector<int>gener(int n) {
	vector<int> arr(n);
	random_device rd;
	mt19937 gen(rd());
	uniform_int_distribution<int>
		dist(1, 1000);
	for (int i = 0; i < n; i++) {
		arr[i] = dist(gen);
	}
	return arr;
}
vector<vector<int>>copy(const vector<vector<int>>& original) {
	vector<vector<int>>copy;
	for (const auto& innerArray : original) {
		copy.push_back(innerArray);
	}
	return copy;
}
